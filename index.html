1. Line plot using sample data¶

import seaborn as sns
import matplotlib.pyplot as plt

x = [1, 2, 3, 4, 5]
y = [10, 15, 13, 18, 22]

sns.lineplot(x=x, y=y)
plt.xlabel('X-axis')
plt.ylabel('Y-axis')
plt.title('Line Plot with Seaborn')
plt.show()

import matplotlib.pyplot as plt
import numpy as np

data = np.random.randn(1000)

sns.histplot(data, bins=30, color='blue', alpha=0.7)
plt.xlabel('Value')
plt.ylabel('Frequency')
plt.title('Histogram with Seaborn')
plt.show()

5. Box plot using iris dataset with sepal length and species on y and x axes¶

In [5]:
import seaborn as sns
import matplotlib.pyplot as plt

data = sns.load_dataset('iris')

sns.boxplot(x='species', y='sepal_length', data=data)
plt.xlabel('Species')
plt.ylabel('Sepal Length')
plt.title('Box Plot with Seaborn')
plt.show()
